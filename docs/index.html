library(brms)

## Loading required package: ggplot2

x <-  seq(1:100)
y <- x + rnorm(100,0,10)
d <- data.frame(x=x, y=y)
plot(x,y)

fit <- brm(y ~ x, 
           data = d,
           cores = 3, chains = 3)
## Compiling the C++ model
## Start sampling
summary(fit)
